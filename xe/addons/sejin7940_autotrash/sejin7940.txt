function triggerDeleteDocumentComments(&$obj) {
		if(Context::get('trash_delete')!='Y') {
			$document_srl = $obj->document_srl;
			if(!$document_srl) return new Object();

			return $this->deleteComments($document_srl, $obj);
		}
}
